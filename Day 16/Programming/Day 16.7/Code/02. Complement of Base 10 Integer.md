**C++ Code :**
```c++
  int bitwiseComplement(int n) {
        if(n == 0){
            return 1;
        }
        int ans = 0;
        int k =1;
        
        while(n !=0){
            if((n&1)==0){
                ans = ans + k;
            }
            k = k<<1;
            n = n>>1;

        }
        return ans;
    }
```
